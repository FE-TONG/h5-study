var myBtn1 = document.getElementById("btn1");
function playPause(){
   if(myVideo.paused){
     myVideo.play();
     myBtn1.style.background="url(images/Pause.png) no-repeat";
   }else{
     myVideo.pause();
     myBtn1.style.background="url(images/Play.png) no-repeat";
   }
}
